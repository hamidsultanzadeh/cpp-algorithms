#include <iostream>

using namespace std;
pair<int,int> binarySearch(int arr[],int size,int n){
    
    int left=0, right=size-1;
    int middle;
    
    while(left <= right){
        middle = (left + right) / 2;
        
        if(arr[middle] > n){
            right = middle - 1;
        } else if(arr[middle] < n){
            left = middle + 1;
        } else if(arr[middle] == n){
            return make_pair(arr[middle],middle+1);
        }
    }
    
    return make_pair(-1,-1);
    
}
int main(int argc, char** argv) {
    
    int arr[10]={1,3,5,7,9,11,12,14,15,19};
    
    pair<int,int> result = binarySearch(arr,10,3);
    
    cout<<"n : "<<result.first<<endl<<"i : "<<result.second;
    

    return 0;
}
