#include <bits/stdc++.h>
using namespace std;
int main(int argc, char** argv) {
    
    bitset<10> s(string ("0110011000"));
    bitset<10> s1(string ("1001001011"));
    
    int sum=0;
    
    for(int i = 0 ; i<s.size() ; i++){
        cout<<s[i];
        sum += s[i]*pow(2,i);
    }
    
    cout<<"\nToInt : "<<sum<<"\n";
    
    cout<<(s&s1)<<endl;
    cout<<(s|s1)<<endl;

    return 0;
}
